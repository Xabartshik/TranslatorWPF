// break Ð¸ continue
#include <iostream>
using namespace std;

int main() {
    int i;
    i = 0;
    while (i < 10) {
        if (i == 3) {
            i = i + 1;
            continue;
        }
        if (i == 7) {
            break;
        }
        cout << i << endl;
        i = i + 1;
    }
    return 0;
}
