#include <iostream>
using namespace std;

int main() {
    int n;        // "параметр" рекурсии
    int sum = 0;  // "накопитель" результата

    cout << "Введите n: ";
    cin >> n;

    int state = 0; // 0 - шаг вниз (как рекурсивный вызов), 1 - шаг вверх (как возврат)

    while (true) {
        if (state == 0) {
            // "рекурсивный вызов": спуск вниз
            if (n == 0) {
                // базовый случай: sum = 0
                sum = 0;
                state = 1; // начинаем подниматься вверх
            } else {
                // имитация рекурсивного вызова sum(n-1)
                n = n - 1;
                // остаёмся в state == 0, пока не дойдём до n == 0
            }
        } else {
            // "возврат" из рекурсии: sum = sum + (n+1)
            sum = sum + (n + 1);
            if (n == 0) {
                break; // вышли из самой верхней "функции"
            } else {
                n = n - 1; // возвращаемся выше по "стеку"
            }
        }
    }

    cout << "Сумма ряда = " << sum << endl;
    return 0;
}
